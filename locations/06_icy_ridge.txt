// Level: Icy Ridge

var loadouts = import git/loadouts

func equip_dashing()
  equipL big sword *10
  equipR shield *6 dashing

func equip_staff()
  equip staff *8 fire +4
  
func equip_crossbow()
  equipL crossbow *10 +10
  equipR crossbow *10 +11

var hrminir_encountered = false
  
?loc = icy_ridge
  // Small enemies
  ?foe = elemental | foe = ki
    ?hp = maxhp | foe.distance <= 23
      // Healed completely or enemy too close
      equip_staff()
  :?foe = pillar | foe = ice_wall
    ?foe.distance <= 5
      equip blade_of_god
      ?hrminir_encountered &
      ^ item.GetCooldown("blade_of_god") <= 0
        activate R
    :
      equip_dashing()

  // HrÃ­mnir
  :?foe = boss
    hrminir_encountered = true
    ?foe.distance <= 2
      ?foe.armor > 0
        equip_crossbow()
    :?foe.distance <= 5
      // The shovel has an attack range of 1
      // so it forces the player to move very
      // very close. So close that it can avoid
      // the attacks (snowballs)
      equip shovel
    :?foe.distance > 5
      equip_dashing()

  loadouts.healwalk()
